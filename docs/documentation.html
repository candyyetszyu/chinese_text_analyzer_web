🔧 Port Configuration

Default Port: The application runs on port 3000 by default.

Method 1: Environment Variable (Recommended)

Set the PORT environment variable before running:

# On macOS/Linux:
export PORT=8080
python run_web.py

# On Windows:
set PORT=8080
python run_web.py

Method 2: Edit run_web.py

  1. Open run_web.py
  2. Change this line:
    # Change this line:
    port = int(os.environ.get('PORT', 3000))

    # To your desired port:
    port = int(os.environ.get('PORT', 8080)) # Example: port 8080
  3. Save and run: python run_web.py

Method 3: Direct Flask App Configuration

  1. Open src/web/app.py
  2. Find the bottom of the file and change:
    if __name__ == '__main__':
        app.run(debug=True, port=3000) # Change 3000 to your port

Common Port Conflicts

Port Common Usage Solution
5000 macOS Control Center (AirPlay Receiver) Use port 3000 (default) or 8080
8080 Development servers Use ports 3001, 8081, 9000
3000 React/Node.js apps Use ports 3001, 3002, 8082
Important Notes:
  • Always restart the application after changing the port
  • Update browser URL to match the new port
  • Check firewall settings if accessing from other devices
  • JavaScript API calls will automatically use the correct port

const axios = require('axios');

class ChineseTextAnalyzer {
    constructor(baseURL = 'http://localhost:3000') {
        this.baseURL = baseURL;
    }

    async analyze(text, options = {}) {
        try {
            const response = await axios.post(
                `${this.baseURL}/api/analyze`,
                { text, ...options }
            );
            return response.data;
        } catch (error) {
            throw new Error(`Analysis failed: ${error.message}`);
        }
    }

    async convert(text, direction = 'to_traditional') {
        const response = await axios.post(
            `${this.baseURL}/api/convert`,
            { text, direction }
        );
        return response.data.converted_text;
    }
}

// Usage
const analyzer = new ChineseTextAnalyzer();
analyzer.analyze("測試文本").then(console.log);

☕ Java Example

Java Integration
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.URI;
import java.time.Duration;

public class ChineseTextAnalyzer {
    private final HttpClient client;
    private final String baseUrl;

    public ChineseTextAnalyzer(String baseUrl) {
        this.baseUrl = baseUrl;
        this.client = HttpClient.newBuilder()
            .connectTimeout(Duration.ofSeconds(10))
            .build();
    }

    public String analyzeText(String text) throws Exception {
        String json = String.format(
            "{\"text\":\"%s\",\"advanced_viz\":[\"pie\",\"vertical\"]}",
            text.replace("\"", "\\\""))
        );

        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(baseUrl + "/api/analyze"))
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers.ofString(json))
            .build();

        HttpResponse<String> response = client.send(request,
            HttpResponse.BodyHandlers.ofString());

        return response.body();
    }

    // Usage
    public static void main(String[] args) throws Exception {
        ChineseTextAnalyzer analyzer = new ChineseTextAnalyzer("http://localhost:3000");
        String result = analyzer.analyzeText("測試文本");
        System.out.println(result);
    }
}

⚙️ Error Handling & Best Practices

Error Handling
# Python error handling example
import requests
from requests.exceptions import RequestException

def safe_analyze(text, max_retries=3):
    for attempt in range(max_retries):
        try:
            response = requests.post(
                "http://localhost:3000/api/analyze",
                json={"text": text},
                timeout=30
            )
            response.raise_for_status()
            return response.json()

        except RequestException as e:
            print(f"Attempt {attempt + 1} failed: {e}")
            if attempt == max_retries - 1:
                raise
            time.sleep(2 ** attempt) # Exponential backoff
Best Practices
  • Timeout: Set reasonable timeouts (30s+)
  • Retry Logic: Implement exponential backoff
  • Validation: Validate input text before sending
  • Async: Use async calls for large texts
  • Caching: Cache results for repeated texts
  • Monitoring: Monitor API response times
Rate Limit: 100 requests/minute per IP
Integration Tips
  • Production: Use environment variables for API URLs

Adding Custom Sentiment Model
# src/core/ml_models.py
import torch
from transformers import AutoTokenizer, AutoModelForSequenceClassification

class ChineseSentimentModel:
    def __init__(self, model_name="hfl/chinese-roberta-wwm-ext"):
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForSequenceClassification.from_pretrained(model_name)
        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        self.model.to(self.device)

    def predict(self, text):
        inputs = self.tokenizer(text, return_tensors="pt", truncation=True)
        inputs = {k: v.to(self.device) for k, v in inputs.items()}

        with torch.no_grad():
            outputs = self.model(**inputs)
            predictions = torch.nn.functional.softmax(outputs.logits, dim=-1)

        return {
            'negative': predictions[0][0].item(),
            'positive': predictions[0][1].item()
        }

# Custom NER model integration
from transformers import AutoTokenizer, AutoModelForTokenClassification
from transformers import pipeline

class ChineseNERModel:
    def __init__(self):
        self.ner_pipeline = pipeline(
            "ner",
            model="ckiplab/bert-base-chinese-ner",
            tokenizer="ckiplab/bert-base-chinese-ner",
            aggregation_strategy="simple"
        )

    def extract_entities(self, text):
        entities = self.ner_pipeline(text)
        processed_entities = {}

        for entity in entities:
            label = entity['entity_group']
            word = entity['word']
            confidence = entity['score']

            if label not in processed_entities:
                processed_entities[label] = []
            processed_entities[label].append({
                'text': word,
                'confidence': confidence
            })

        return processed_entities
